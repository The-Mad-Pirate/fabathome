00040 //----[  allPatherImplementations  ]-------------------------------------------
00041 QStringList PatherInterface::allPatherImplementations() {
00042   QStringList pather_impls;
00043 
00044   int i = 0;
00045   PatherImplementation impl = getImplementation(0);
00046   while (impl != kInvalidPatherImplementation) {
00047     pather_impls.push_back(getPatherName(impl));
00048     impl = getImplementation(++i);
00049   }
00050 
00051   return pather_impls;
00052 }

00054 //----[  getPatherName  ]------------------------------------------------------
00055 QString PatherInterface::getPatherName(PatherImplementation implementation) {
00056   switch (implementation) {
00057   case kSimpleCrosshatchPather: return "Simple Crosshatch Pather";
00058   case kDoubleEdgePather:       return "Double Edge Pather";
00059   case kEdgeOnlyDebugPather:    return "Edge-Only Pather";
00060   case kConcentricLoopsPather:  return "Concentric Loops Pather (Theoretical; Don't Use)";
00061   default:
00062   case kInvalidPatherImplementation:  return "(Invalid Pather)";
00063   }
00064 }

00066 //----[  getImplementation  ]--------------------------------------------------
00067 PatherImplementation PatherInterface::getImplementation(
00068     int implementation_index) {
00069   static const PatherImplementation IMPLEMENTATIONS[] = {
00070     kSimpleCrosshatchPather,
00071     kDoubleEdgePather,
00072     kEdgeOnlyDebugPather,
00073     kConcentricLoopsPather,
00074   };
00075   static const int NUMBER_OF_IMPLEMENTATIONS
00076       = sizeof(IMPLEMENTATIONS) / sizeof(PatherImplementation);
00077   if (implementation_index < 0 ||
00078       implementation_index >= NUMBER_OF_IMPLEMENTATIONS) {
00079     return kInvalidPatherImplementation;
00080   } else {
00081     return IMPLEMENTATIONS[implementation_index];
00082   }
00083 }

00104 //----[  makePather  ]---------------------------------------------------------
00105 Pather* PatherInterface::makePather(PatherImplementation impl) {
00106   Pather* pather;
00107 
00108   switch (impl) {
00109   case kSimpleCrosshatchPather:
00110     pather = new SimpleCrossHatchPather();
00111     break;
00112   case kDoubleEdgePather:
00113     pather = new DoubleEdgePather();
00114     break;
00115   case kEdgeOnlyDebugPather:
00116     pather = new EdgeOnlyPather();
00117     break;
00118     case kConcentricLoopsPather:
00119   pather = new ConcentricLoopsPather();
00120     break;
00121   default:
00122     return NULL;
00123   }
00124 
00125   return pather;
00126 }
