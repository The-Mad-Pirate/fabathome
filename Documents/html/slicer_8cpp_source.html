00095                        SlicerProgressCallback* callback) {

00099 
00100   // problems with this in the past
00101   //jlz27 - does not work in Qt 4.6.2
00102   //confirm (_finite(slice_height) && !_isnan(slice_height)) else return false;

00104   QList<FAHTriangle> triangles;
00105   input_region->getTriangles(&triangles);
00106 
00107   if (triangles.isEmpty()) {
00108     if (callback) {
00109       callback->encounteredIssue("Region was empty");
00110     }
00111     return true;
00112   }

00114   FAHVector3 min, max;
00115   min.set(triangles.at(0).v[0]);
00116   max.set(triangles.at(0).v[0]);
00117   for (int i = 0; i < triangles.size(); ++i) {
00118     for (int j = 0; j < 3; ++j) {
00119       const FAHTriangle& triangle = triangles.at(i);
00120       min.min(triangle.v[j]);
00121       max.max(triangle.v[j]);
00122     }
00123   }
00124

00126 
00127   // This value is set to put the first slice at z = slice height
00128   // no matter where the alignment is specified to occur
00129   float z_adjustment;
00130 
00131   switch(alignment) {
00132   default:
00133     confirm(!"unknown value for alignment; defaulting to 'tray' alignment");
00134 
00135   case kAlignTray: {
00136     z_adjustment = 0.0f;
00137     FAHFloat z = slice_height;
00138     while (z < max.z) {
00139       slice_z.push_back(z);
00140       z += slice_height;
00141     }
00142     } break;
00143 
00144   case kAlignTop: {
00145     FAHFloat z = max.z - slice_height;
00146     while (z > min.z) {
00147       slice_z.push_front(z);
00148       z -= slice_height;
00149     }
00150     z_adjustment = slice_height-z;
00151   } break;
00152 
00153   case kAlignBottom: {
00154     FAHFloat z = min.z + slice_height;
00155     z_adjustment = slice_height-z;
00156     while (z < max.z) {
00157       slice_z.push_back(z);
00158       z += slice_height;
00159     }
00160   } break;
00161 
00162   case kAlignCenter: {
00163     FAHFloat z = (min.z + max.z) / 2;
00164     while (z > min.z) z -= slice_height;
00165     z_adjustment = slice_height-z;
00166     z += slice_height;
00167     while (z < max.z) {
00168       slice_z.push_back(z);
00169       z += slice_height;
00170     }
00171   } break;
00172   }
00173 
00174   if (callback) callback->setEstimatedNumberOfSlices(slice_z.size());
00175 
00176   for (int slice_index = 0; slice_index < slice_z.size(); ++slice_index) {
00177     QList<FAHLine> segments;
00178     const float current_z = slice_z.at(slice_index);
00179 
00180     // Form a list of all of the line segments in the slice by
00181     // intersecting the plane with triangles.
00182     todo("kwg8", "make this intersection test not check every single triangle in the source mesh");
00183     for (int j=0; j < triangles.size(); ++j) {
00184       FAHLine line;
00185       if (triangles.at(j).intersectXYPlaneAtZ(current_z, &line)) {
00186         line.a.z += z_adjustment;
00187         line.b.z += z_adjustment;
00188         segments.append(line);
00189       }
00190     }
00191 
00192     // no intersections? skip this plane.
00193     if (segments.isEmpty()) {
00194       QString description;
00195       QTextStream(&description) << "warning: slice " << slice_index << " is empty\n";
00196       if (callback) callback->encounteredIssue(description);
00197       continue;
00198     }
00199 
00200     // Discover all of the loops in the segments
00201     QVector<FAHLoopInXYPlane> loops;
00202     findLoops(segments, &loops);
00203 
00204     // pull out all of the outer boundaries into regions
00205     QVector<SliceMaterialRegion> regions;
00206     for (int i = loops.size() - 1; i >= 0; --i) {
00207       const FAHLoopInXYPlane& loop = loops.at(i);
00208       switch (loop.sense()) {
00209       case FAHLoopInXYPlane::kCounterClockwise: {
00210         // add as the outer boundary of a region
00211         SliceMaterialRegion region;
00212         region.setOuterBoundary(loops.at(i));
00213         loops.remove(i);
00214         regions.append(region);
00215       } break;
00216       case FAHLoopInXYPlane::kClockwise:
00217         // don't do anything with this yet
00218         break;
00219       default:
00220       case FAHLoopInXYPlane::kUndefined:
00221         // discard this malformed loop
00222         loops.remove(i);
00223         if (callback) {
00224           QString issue_text;
00225           QTextStream(&issue_text) << "malformed loop removed";
00226           callback->encounteredIssue(issue_text);
00227         }
00228         break;
00229       }
00230     }
00231 
00232     // for each of the inner boundaries, find the tightest
00233     // outer boundary that contains that inner boundary
00234     // and make this inner boundary a child of that region
00235     for (int i = loops.size() - 1; i >= 0; --i) {
00236       const FAHLoopInXYPlane& loop = loops.at(i);
00237       int tightest_container_index = -1;
00238       for (int j = 0; j < regions.size(); ++j) {
00239         if (regions.at(j).getOuterBoundary().contains(loop) &&
00240             (tightest_container_index < 0 ||
00241              regions.at(tightest_container_index).getOuterBoundary().contains(regions.at(j).getOuterBoundary()))) {
00242           tightest_container_index = j;
00243         }
00244       }
00245       // we should always have a container for an inner
00246       // boundary, otherwise something is wrong with the
00247       // model.
00248       if (tightest_container_index >= 0) {
00249         regions[tightest_container_index].addInnerBoundary(loop);
00250       } else {
00251         if (callback) {
00252           QString issue_text;
00253           QTextStream(&issue_text) << "Inner-boundary loop was uncontained by an outer boundary";
00254           callback->encounteredIssue(issue_text);
00255         }
00256       }
00257     }
00258 
00259     // create the regions in the output material
00260     SliceMaterial material(input_region->getFillMaterialId());
00261     for (int i = 0; i < regions.size(); ++i) {
00262       material.addSliceMaterialRegion(regions.at(i));
00263     }
00264 
00265     // build the output slice
00266     Slice slice(current_z + z_adjustment);
00267     slice.addSliceMaterial(material);
00268     slicestack.addSlice(slice);
00269 
00270     // report that computation of this slice is complete
00271     if (callback) callback->sliceHasBeenComputed(slice_index, current_z);
00272   }
00273 
00274   // all of the slices have been finished
00275   if (callback) callback->finishedSlicing(slicestack);
00276 
00277   return true;
00278 }

00302 void Slicer::findLoops(const QList<FAHLine>& segments,
00303                        QVector<FAHLoopInXYPlane>* loops_out) {
00304 
00305   // Start with some minimum tolerance
00306   FAHFloat tolerance = 1e-8;
00307   const FAHFloat kHighestTolerance = 1e-2;
00308   QVector<FAHLoopInXYPlane> open_loops;
00309   QVector<FAHLoopInXYPlane>& closed_loops = *loops_out;
00310 
00311   // Add all segments as "open" loops
00312   foreach (const FAHLine& segment, segments) {
00313     FAHLoopInXYPlane loop;
00314     loop.add(segment.a);
00315     loop.add(segment.b);
00316     open_loops.append(loop);
00317   }
00318 
00319   // Try to merge open loops
00320   while (tolerance <= kHighestTolerance && !open_loops.isEmpty()) {
00321     // For each loop, try to merge it with other open loops
00322     for (int k = open_loops.size() - 1; k >= 0; --k) {
00323       FAHLoopInXYPlane& k_loop = open_loops[k];
00324       const FAHVector3& k_first = k_loop.points.first();
00325       const FAHVector3& k_last = k_loop.points.last();
00326       for (int j = k - 1; j >= 0; --j) {
00327         FAHLoopInXYPlane& j_loop = open_loops[j];
00328         const FAHVector3& j_first = j_loop.points.first();
00329         const FAHVector3& j_last = j_loop.points.last();
00330 
00331         // Are any two of the loops' endpoints coincident?
00332         if (k_first.distanceToSq(j_first) < tolerance) {
00333           // prepend the reversed list of points in j, except the first point
00334           for (int i = 1; i < j_loop.points.size(); ++i) {
00335             k_loop.addFirst(j_loop.points.at(i));
00336           }
00337         } else if (k_first.distanceToSq(j_last) < tolerance) {
00338           // prepend the list of points in j, except
00339           // for the last point
00340           for (int i = j_loop.points.size() - 2; i >= 0; --i) {
00341             k_loop.addFirst(j_loop.points.at(i));
00342           }
00343         } else if (k_last.distanceToSq(j_first) < tolerance) {
00344           // append the list of points in j, except the first point
00345           for (int i = 1; i < j_loop.points.size(); ++i) {
00346             k_loop.points.append(j_loop.points.at(i));
00347           }
00348         } else if (k_last.distanceToSq(j_last) < tolerance) {
00349           // append the reversed list of points in j, except the last point
00350           for (int i = j_loop.points.size() - 2; i >= 0; --i) {
00351             k_loop.points.append(j_loop.points.at(i));
00352           }
00353         } else {
00354           // these two loops do not connect
00355           continue;
00356         }
00357 
00358         // Has this loop been completed?
00359         if (k_loop.points.first().distanceToSq(k_loop.points.last()) < tolerance) {
00360 
00361           // Remove the duplicate point in this loop
00362           k_loop.points.remove(0);
00363 
00364           // The loop is now closed
00365           closed_loops.append(k_loop);
00366           open_loops.remove(k);
00367         }
00368 
00369         // We merged the j'th loop into the k'th one
00370         open_loops.remove(j);
00371 
00372         // Restart examining the list of open loops
00373         k = open_loops.size();
00374         j = 0;
00375         break;
00376       }
00377     }
00378 
00379     // If there are any outstanding open loops, the tolerance
00380     // might be too low to close the list; so, increase the
00381     // tolerance
00382     tolerance *= 2;
00383   }
00384 
00385   // Any remaining open loops are invalid.  However, to make the
00386   // best of the situation, we'll discard only those that are
00387   // truly invalid (those with only 2 points) and try to keep
00388   // the rest.
00389   if (!open_loops.isEmpty()) {
00390     QString issue_text;
00391     QTextStream(&issue_text) << open_loops.size()
00392                              << " unclosed loop"
00393                              << (open_loops.size() != 1 ? "s" : "")
00394                              << "; treating { ";
00395     bool is_first = true;
00396     for (int k = open_loops.size() - 1; k >= 0; --k) {
00397       if (open_loops.at(k).points.size() <= 2) {
00398         // remove this degenerate loop
00399         open_loops.remove(k);
00400       } else {
00401         if (is_first) {
00402           QTextStream(&issue_text) << open_loops.at(k).points.size();
00403         } else {
00404           QTextStream(&issue_text) << "," << open_loops.at(k).points.size();
00405         }
00406         is_first = false;
00407       }
00408     }
00409     QTextStream(&issue_text) << " } points as closed";
00410     if (callback) callback->encounteredIssue(issue_text);
00411     closed_loops += open_loops;
00412   }
00413 
00414   // Simplify all of the loops
00415   for (int i = closed_loops.size() - 1; i >= 0; --i) {
00416     closed_loops[i].simplify();
00417     if (closed_loops[i].sense() == FAHLoopInXYPlane::kUndefined) {
00418       closed_loops.remove(i);
00419     }
00420   }
