00047 //----[  doPathing  ]----------------------------------------------------------
00048 bool Pather::doPathing(const SliceStack& slices,
00049                        PatherProgressCallback* callback) {
00050   const SliceArray& slice_array = slices.getSlices();
00051 
00052   // Let the callback know how many slices are going to be computed
00053   if (callback) callback->setNumberOfSlices(slice_array.size());
00054 
00055   for (int i = 0; i < slice_array.size(); i++) {
00056 
00057     // Start building a new slice
00058     PathSlice* current_slice = new PathSlice(slice_array[i].getZ());
00059 
00060     // Put the paths for all of the materials into the slice
00061     const SliceMaterialArray& material_array = slice_array[i].getMaterials();
00062     for (int j = 0; j < material_array.size(); j++) {
00063 
00064       // Build this material
00065       PathSliceMaterial* current_material
00066           = new PathSliceMaterial(material_array[j].getMaterialID());
00067 
00068       // For each of the different regions this material contains in this
00069       // layer, add all of the paths that are needed to cover that region.
00070       const SliceMaterialRegionArray& regions = material_array[j].getRegions();
00071       for (int k = 0; k < regions.size(); ++k) {
00072         PathSliceRegion* current_region = new PathSliceRegion();
00073         if (!runPathingAlgorithm(regions[k], current_region, callback)) {
00074           if (callback) callback->encounteredIssue(getLastErrorString());
00075           return false;
00076         }
00077         current_material->add(current_region);
00078       }
00079 
00080       current_slice->addMaterial(current_material);
00081     }
00082     path_stack_.addSlice(current_slice);
00083     if (callback) callback->sliceHasBeenComputed(i, current_slice);
00084   }
00085 
00086   // successs!
00087   return true;
00088 }
