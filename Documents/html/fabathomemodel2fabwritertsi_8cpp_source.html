00059 void FabAtHomeModel2FabWriterTSI::addMeshes(QString material_name,
00060                                             QScriptValue material_calibration,
00061                                             QScriptValue meshes_array) {
00062   MaterialProperties* material_properties = new MaterialProperties();
00063 
00064   { // iterate over material calibration properties
00065     QScriptValueIterator iterator(material_calibration);
00066     while (iterator.hasNext()) {
00067       iterator.next();
00068       material_properties->insert(iterator.name(),
00069                                   iterator.value().toString());
00070     }
00071 
00072     material_properties->insert("name", material_name);
00073 
00074     materials_list_.append(material_properties);
00075   }
00076 
00077   { // save the meshes' stacks into the main list of stacks, paired
00078     // with this material calibration
00079     QScriptValueIterator iterator(meshes_array);
00080 
00081     while (iterator.hasNext()) {
00082       iterator.next();
00083       QScriptValue value = iterator.value();
00084       if (!value.isQObject()) continue;
00085       QObject* qo = value.toQObject();
00086       AMFRegionTSI* region_tsi =
00087           (AMFRegionTSI*)(0 == QString::compare("AMFRegionTSI",
00088                                                 qo->metaObject()->className()) ? qo : NULL);
00089       PrintableRegionPair p(material_properties,
00090                             region_tsi);
00091       regions_.append(p);
00092     }
00093   }
