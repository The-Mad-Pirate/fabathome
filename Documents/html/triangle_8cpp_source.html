00122 bool Triangle::intersectXYPlaneAtZ(Float z, Line* intersection) const {
00123 
00124   if ((v[0].z >= z && v[1].z >= z && v[2].z >= z) ||
00125       (v[0].z <= z && v[1].z <= z && v[2].z <= z)) {
00126     return false;
00127   }
00128 
00129   Vector3 p, p1, p2;
00130   int points_found = 0;
00131 
00132   for (int i0=0; i0<3; i0++) {
00133     int i1 = (i0+1)%3;
00134     if ((v[i0].z <= z && v[i1].z > z) || (v[i1].z <= z && v[i0].z > z)) {
00135       p.set(v[i0]).add(v[i1].copy().sub(v[i0]).scale((z - v[i0].z)/(v[i1].z-v[i0].z)));
00136       //p = v[i0]+(v[i1]-v[i0])*(z-v[i0].z)/(v[i1].z-v[i0].z);
00137       if (points_found == 0) {
00138         p1.set(p);
00139         points_found++;
00140       } else {
00141         p2.set(p);
00142         // set right order
00143         if ((normal().cross(p2.copy().sub(p1))).dot(Vector3::make(0,0,1)) < 0) { // swap
00144           p.set(p1);
00145           p1.set(p2);
00146           p2.set(p);
00147         }
00148         intersection->set(p1, p2);
00149         return true;
00150       }
00151     }
00152   }
00153 
00154   return false; // less than 2 points found
00155 }

00159 bool Triangle::intersectedByRay(const Vector3& origin,
00160                                 const Vector3& ray) {
00161   Vector3 e1,e2;
00162   e1.set(v[1]).sub(v[0]);
00163   e2.set(v[2]).sub(v[0]);
00164 
00165   Vector3 q;
00166   q.set(ray).cross(e2);
00167   double a=e1.dot(q);
00168 
00169   // vector is parallel to plane
00170   if (floatsEqual(a, 0)) return false;
00171 
00172   double f=1/a;
00173   Vector3 s;
00174   s.set(origin).sub(v[0]);
00175   double u=f*s.dot(q);
00176 
00177   // the intersection is outside the triangle
00178   if (u<0.0) return false;
00179 
00180   Vector3 r;
00181   r=s.cross(e1);
00182   double v=f*ray.dot(r);
00183 
00184   // the intersection is outside the triangle
00185   if (v<0.0 || u+v>1.0) return false;
00186 
00187   return true;
