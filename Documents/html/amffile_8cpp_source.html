00053 bool AMFFile::readFromFile(QString path) {
00054 
00055   // Remove anything currently in this structure
00056   clear();
00057 
00058   QDomDocument doc;
00059   QFile file(path);
00060 
00061   if (!file.open(QFile::ReadOnly)) {
00062     return false; //error
00063   }
00064 
00065   doc.setContent(&file);
00066 
00067   // xml in memory, can release file
00068   file.close();
00069 
00070   QDomElement root = doc.documentElement();
00071   if (0 != QString::compare(root.tagName(), "amf", Qt::CaseInsensitive)) {
00072     return false; //error
00073   }
00074 
00075   // Get all the object nodes
00076   QDomNodeList objectNodes = root.elementsByTagName("Object");
00077 
00078   for (unsigned int i = 0; i < objectNodes.length(); i++) {
00079     AMFObject* newAmfObj = new AMFObject();
00080     objects.append(newAmfObj);
00081     QDomElement objectElement = objectNodes.at(i).toElement();
00082     newAmfObj->ReadFromXml(objectElement);
00083   }
00084 
00085   return true;
00086 }

00088 bool AMFFile::writeToFile(QString path) {
00089   QDomDocument doc;
00090   QDomElement root = doc.createElement("AMF");
00091   doc.appendChild(root);
00092 
00093   // Write each object to the xml

00095     QDomElement obj = doc.createElement("Object");
00096     objects.at(i)->WriteXml(obj);
00097     root.appendChild(obj);
00098   }
00099 
00100   QFile file(path);
00101   QTextStream outStream(&file);
00102 
00103   if (!file.open(QFile::WriteOnly))
00104     return false; //error
00105 
00106   outStream << "<!DOCTYPE AMF>\n";
00107   outStream << doc.toString(3);
00108 
00109   file.close();
00110 
00111   return true;
