00059 void PrintThread::run() {

00065   connect(&progress_tsi, SIGNAL(setProgress(int)), this, SLOT(emitSetMasterProgress(int)), Qt::QueuedConnection);

00069   connect(&slicer_callback, SIGNAL(signalSetTaskProgress(int)), this, SLOT(emitSetTaskProgress(int)), Qt::QueuedConnection);

00072   connect(&slicer_callback, SIGNAL(signalSetProcessingPlane(float)), this, SLOT(emitSetProcessingPlane(float)), Qt::QueuedConnection);

00074   connect(&pather_callback, SIGNAL(signalSetTaskRange(int,int)), this, SLOT(emitSetTaskRange(int,int)), Qt::QueuedConnection);
00075   connect(&pather_callback, SIGNAL(signalSetTaskProgress(int)), this, SLOT(emitSetTaskProgress(int)), Qt::QueuedConnection);
00076   connect(&pather_callback, SIGNAL(signalEncounteredIssue(QString)), this, SLOT(emitLogEntry(QString)), Qt::QueuedConnection);
00077   connect(&pather_callback, SIGNAL(signalSetTaskTitle(QString)), this, SLOT(emitSetTaskTitle(QString)), Qt::QueuedConnection);
00078   connect(&pather_callback, SIGNAL(signalAddRenderablePath(int,Path*)), this, SLOT(emitAddRenderablePath(int,Path*)), Qt::QueuedConnection);
00079 
00080   // Get a set of <tool name, geometry> pairs
00081   QMultiMap<QString,AMFRegion*> meshes = main_window_->getMaterialNameToPrintableMeshMap();
00082   if (meshes.isEmpty()) return; // Make sure geometry exists!
00083   tool_script_->remapMaterialKeysToToolKeys(&meshes);
00084 
00085 
00086 
00087   // Automatically generate support material
00088   AMFMesh support_material_mesh;
00089   if (generate_support_material_) {
00090     QList<AMFRegion*> printing_regions = meshes.values();
00091     QString tool_name;
00092     if (tool_script_->getSupportMaterialTool(&tool_name) && // inexpensive; do it before requires...() check
00093         requiresSupportMaterial(printing_regions)) {
00094 
00095       // Create the supporting mesh for this object, and add its geometry
00096       // to the printing map
00097       generateSupportingMesh(printing_regions, &support_material_mesh);
00098       foreach (AMFRegion* region, *support_material_mesh.GetRegions()) {
00099         meshes.insertMulti(tool_name, region);
00100       }
00101     }
00102   }
00103 
00104   // create a new set of meshes
00105   QList<AMFMesh*> allocated_meshes;
00106   { // Regenerate all meshes
00107     QMap<QString,int> tool_materials;
00108     foreach (const QString& tool, meshes.keys()) {
00109       if (!tool_materials.contains(tool)) {
00110         int id = tool_materials.size();
00111         internal_material_id_to_color_.insert(id, QColor((Qt::GlobalColor)(int(Qt::red)+id)));
00112         tool_materials.insert(tool, id);
00113       }
00114     }
00115 
00116     // Get the set of meshes
00117     QSet<AMFMesh*> amf_meshes;
00118     QMultiMap<AMFMesh*,AMFRegion*> regions;
00119     QMap<AMFRegion*,QString> region_tools;
00120     for (QMultiMap<QString,AMFRegion*>::const_iterator i = meshes.begin(); i != meshes.end(); ++i) {
00121       AMFRegion* region = i.value();
00122       region_tools.insert(region, i.key());
00123       amf_meshes.insert(region->getParent());
00124       regions.insertMulti(region->getParent(), region);
00125     }
00126     meshes.clear();
00127 
00128     foreach (AMFMesh* mesh, amf_meshes) {
00129 
00130       QList<AMFVertex*> vertices;
00131       foreach (AMFVertex* vertex, *mesh->GetVertices()) {
00132         vertices.append(new AMFVertex(vertex->getPoint(), vertex->getVertexID()));
00133       }
00134 
00135       AMFMesh* new_mesh = new AMFMesh();
00136       foreach (AMFRegion* region, regions.values(mesh)) {
00137         QString string = region_tools.find(region).value();
00138         AMFRegion* new_region = new AMFRegion(new_mesh);
00139         QList<AMFTriangle*> triangles;
00140         foreach (AMFTriangle* triangle, region->getTriangles()) {
00141           triangles.append(new AMFTriangle(triangle));
00142         }
00143         new_region->setFillMaterialID(tool_materials.find(string).value());
00144         new_region->FillTriangles(triangles);
00145         meshes.insertMulti(string, new_region);
00146       }
00147 
00148       new_mesh->setVertices(vertices);
00149       allocated_meshes.append(new_mesh);
00150     }
00151   }
00152 
00153   QString fab_file;
00154   if (tool_script_->run(meshes,
00155                         &slicer_callback,
00156                         pather_implementation_,
00157                         &pather_callback,
00158                         &progress_tsi,
00159                         &fab_file)) {
00160     emit setTaskTitle("Finished!");
00161     emit finishedFabFile(fab_file);
00162   } else {
00163     emit setTaskTitle("Printing error; see output (advanced mode) for details");
00164   }
00165 
00166   for (int i = 0; i < allocated_meshes.size(); ++i) {
00167     delete allocated_meshes.at(i);
00168   }
00169 }
