00241 bool AMFRegion::validateEdge(QMap<EdgeKey, bool> *visitedEdges,
00242                              QMap<EdgeKey, bool> *doneEdges,
00243                              int v1, int v2) const {
00244 
00245   EdgeKey edgeKey(v1, v2);
00246   EdgeKey revEdgeKey(v2, v1);
00247 
00248   // edge already exists as a loner - FAIL
00249   if (visitedEdges->contains(edgeKey)) {
00250     //qDebug() << "edge already exists " << v1 << " " << v2;
00251     return false;
00252   }
00253 
00254   if (visitedEdges->contains(revEdgeKey)) {
00255 
00256     // edge and partner have been seen previously - FAIL
00257     if (doneEdges->contains(edgeKey)) {
00258       //qDebug() << "pair already exists " << v1 << " " << v2;
00259       return false;
00260     }
00261 
00262     // record having visited edge and partner
00263     doneEdges->insert(edgeKey, true);
00264     doneEdges->insert(revEdgeKey, true);
00265     visitedEdges->remove(edgeKey);
00266     visitedEdges->remove(revEdgeKey);
00267     //qDebug() << "recording pair " << v1 << " " << v2;
00268   }
00269   else {
00270     visitedEdges->insert(edgeKey, true);
00271     //qDebug() << "recording edge " << v1 << " " << v2;
00272   }
00273 
00274   return true;
