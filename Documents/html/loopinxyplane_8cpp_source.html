00179   // out to infinity and count the number of segments it intersects.
00180   Line test_line(point, max().scale((Float)1.1));
00181   int intersections = 0;

00183     Line current_segment(points[i-1], points[i%points.size()]);

00185     // find an intersection that is not the endpoint of the current segment.
00186     // this ensures that lines that go through a vertex don't get double-counted
00187     if (current_segment.intersectSegmentWithSegment2DXY(test_line, &pt) &&
00188         !current_segment.b.equals(pt)) {

00219   // 1. Add input segment to worklist
00220   // 2. Create an empty finalized list
00221   // 3. Get the next element from the worklist
00222   // 4. For each edge in the loop, test for intersection with
00223   //    the current element.
00224   // 5. If there was an intersection and the point of intersection
00225   //    is not one of the segment's end-points, split the segment
00226   //    at the point of intersection and add both to the worklist.
00227   // 6. If there was no such intersection, add the segment to
00228   //    the finalized list.
00229   // 7. Go to (2) if the worklist still has elements in it
00230   // 8. For each entry in the finalized list, test its midpoint to see
00231   //    if it is inside the loop.  If it is not, discard the segment.
00232 
00233   QVector<Line> worklist;
00234   worklist.append(segment);
00235   inner_segments->clear();
00236 
00237   while (worklist.isEmpty() == false) {
00238     const Line& current_segment = worklist.at(0);
00239     bool intersected = false;
00240     for (int i = 1; i < points.size() + 1; ++i) {
00241       Line loop_segment(points[i-1], points[i%points.size()]);
00242       Vector3 intersection;
00243       intersected = current_segment.intersectSegmentWithSegment2DXY(loop_segment, &intersection);
00244       intersection.z = current_segment.a.z;
00245       intersected = intersected && !current_segment.isEndpoint(intersection);
00246       if (intersected) {
00247         // split the segment in two
00248         Line first(current_segment.a, intersection);
00249         Line second(intersection, current_segment.b);
00250         worklist.append(first);
00251         worklist.append(second);

00256       // this segment is finalized
00257       inner_segments->append(current_segment);
00258     }
00259     worklist.remove(0);

00261 
00262   // Iterate through and remove segments that are not inside the loop.
00263   // Go backward so that removing elements doesn't change our iterator.
