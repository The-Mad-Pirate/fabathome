00090 bool Line::pointOnLine(const Vector3& v) const {
00091   // a + ray(from:a, to:v) * scalar = b
00092   // ray(from:a, to:v) * scalar = b - a
00093   // abs(ray(from:a, to:v)) = abs(ray(from:a, to:b))
00094   Vector3 t0, t1;
00095   return a.equals(v) ||
00096          b.equals(v) || 
00097          t0.ray(a, v).abs().equals(t1.ray(a, b).abs());
00098 }

00101 bool Line::pointOnSegment(const Vector3& v) const {
00102   // the point is in the segment if a ray from a->v is the reverse of
00103   // the ray from b->v, or either endpoint is equal to the point.
00104   bool point_on_segment = a.equals(v) || b.equals(v);
00105   if (!point_on_segment) {
00106     Vector3 t0, t1;
00107     float dot_product = t0.ray(a, v).dot(t1.ray(b, v));
00108     point_on_segment = floatsEqual(dot_product, -1.0);
00109   }
00110   return point_on_segment;
00111 }

00155 Line& Line::changeLength(Float amount) {
00156   confirm(hasLength()) else {
00157     // can't do anything if there is no length, since we can't find a
00158     // direction in which to extend the line
00159     return *this;
00160   }
00161   amount /= Float(2.0);
00162   Vector3 v(ray());
00163   v.normalize();
00164   a.sub(v.copy().scale(amount)); // a += norm(b-a) * amount/2
00165   b.add(v.copy().scale(amount)); // b -= norm(b-a) * amount/2
00166   return *this;
00167 }

00176 bool Line::intersectSegmentWithSegment2DXY(
00177         const Line& other,
00178         Vector3* result) const {
00179 
00180   // algorithm from:
00181   // http://local.wasp.uwa.edu.au/~pbourke/geometry/lineline2d/
00182 
00183   Float ua = (other.b.x - other.a.x)*(a.y - other.a.y)
00184            - (other.b.y - other.a.y)*(a.x - other.a.x);
00185   Float ub = (b.x - a.x)*(a.y - other.a.y) - (b.y - a.y)*(a.x - other.a.x);
00186   Float denom = (other.b.y - other.a.y)*(b.x - a.x)
00187               - (other.b.x - other.a.x)*(b.y - a.y);
00188 
00189   if (denom == Float(0.0)) {
00190     return false; // no intersection; parallel or coincident lines
00191   }
00192 
00193   ua /= denom;
00194   ub /= denom;
00195 
00196   // ensure it is within both segments
00197   if (ua < 0.0 || ua > 1.0 || ub < 0.0 || ub > 1.0) {
00198     return false;
00199   }
00200 
00201   confirm(result) then {
00202     result->set(a.x + ua * (b.x - a.x),
00203                 a.y + ua * (b.y - a.y),
00204                 (a.z+b.z+other.a.z+other.b.z)/4.0);
00205   }
00206   return true;
00207 }

00211 Float Line::evaluate2DLineXY(
00212         const Vector3& point) const {
00213   Float dx = b.x - a.x, dy = b.y - a.y;
00214   // y - y0 = m(x - x0)
00215   // y - m(x - x0) - y0 = 0
00216   if (dx == Float(0.0)) {
00217     // this is an exactly vertical line
00218     Float difference = point.x - a.x;
00219     if (floatsEqual(difference, 0.0)) {
00220       return point.y - /* Float(0.0) -*/ a.y;
00221     } else if (difference > Float(0.0)) {
00222       return /* point.y */- dy * Float(+1e10) /* - a.y */;
00223     } else { //if (difference < Float(0.0)) {
00224       return /* point.y */- dy * Float(-1e10) /* - a.y */;
00225     }
00226   } else {
00227     return (point.y - (dy / dx) * (point.x - a.x) - a.y);
00228   }
00229 }

00232 Float Line::segmentDistanceTo2DXY(const Vector3& point) const {
00233   Float A = point.x - a.x;
00234   Float B = point.y - a.y;
00235   Float C = b.x - a.x;
00236   Float D = b.y - a.y;
00237   Float dot = A*C + B*D;
00238   Float len_sq = C*C + D*D;
00239   Float param = dot / len_sq;
00240 
00241   Float xx, yy;
00242   if (param < (Float)0.0) {
00243     xx = a.x;
00244     yy = a.y;
00245   } else if (param > (Float)1.0) {
00246     xx = b.x;
00247     yy = b.y;
00248   } else {
00249     xx = a.x + param * C;
00250     yy = a.y + param * D;
00251   }
00252 
00253   Float dx = point.x - xx;
00254   Float dy = point.y - yy;
00255 
00256   return sqrt(dx*dx + dy*dy);
00257 }
