00072 Quaternion& Quaternion::set(const Vector3& axis, Float angle) {
00073   confirm(axis.isNormalized()) else {
00074     // the axis must be normalized for this method to work
00075     return axis.isInvalid() ? zero() : set(axis.copy().normalize(), angle);
00076   }
00077   Float sin_by_2 = Float(sin(angle/2));
00078   return set(Float(cos(angle/2)),
00079              axis.x * sin_by_2,
00080              axis.y * sin_by_2,
00081              axis.z * sin_by_2);

00101 Quaternion& Quaternion::setEuler(Float x, Float y, Float z) {
00102   Float c2 = cos(z/2), s2 = sin(z/2);
00103   Float c1 = cos(y/2), s1 = sin(y/2);
00104   Float c3 = cos(x/2), s3 = sin(x/2);
00105   Float c1c2 = c1*c2;
00106   Float s1s2 = s1*s2;
00107   return set(c1c2*c3 - s1s2*s3,
00108              c1c2*s3 + s1s2*c3,
00109              s1*c2*c3 + c1*s2*s3,
00110              c1*s2*c3 - s1*c2*s3);
00111 }

00161 Vector3 Quaternion::eulerAngles() const {
00162   Float sqw = w*w;
00163   Float sqx = x*x;
00164   Float sqy = y*y;
00165   Float sqz = z*z;
00166   Float unit = sqx + sqy + sqz + sqw; // if normalized is one, otherwise is correction factor
00167   Float test = x*y + z*w;
00168   if (test > 0.499*unit) { // singularity at north pole
00169     Vector3 v(0, 2 * atan2(x, w), Math::kPi/2);
00170     return v;
00171   } else if (test < -0.499*unit) { // singularity at south pole
00172     Vector3 v(0, -2 * atan2(x, w), -Math::kPi/2);
00173     return v;
00174   } else {
00175     Vector3 v(atan2(2*x*w-2*y*z, -sqx + sqy - sqz + sqw),
00176               atan2(2*y*w-2*x*z,  sqx - sqy - sqz + sqw),
00177               asin(2*test/unit));
00178     Vector3 f(v);
00179     f.scale(1.0/Math::kPi);
00180     return v;
00181   }
00182 }
