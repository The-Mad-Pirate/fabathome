00063 bool Plane::intersectLine(const Line& line, Vector3* intersection) const {
00064   Vector3 abc(a,b,c);
00065   Float numer = d + abc.dot(line.a);
00066   Float denom = -abc.dot(line.ray());
00067 
00068   if (denom == 0) {
00069     // parallel to the plane
00070     return false;
00071   }
00072 
00073   intersection->set(line.a).add(line.ray().scale(numer/denom));
00074 
00075   return true;
00076 }

00079 bool Plane::intersectSegment(const Line& segment, Vector3* intersection) const {
00080   Vector3 abc(a,b,c);
00081   Float numer = d + abc.dot(segment.a);
00082   Float denom = -abc.dot(segment.ray());
00083 
00084   if (denom == 0) {
00085     // parallel to the plane
00086     return false;
00087   }
00088 
00089   Float t = numer / denom;
00090 
00091   // check to make sure it falls between the segment's endpoints
00092   if (t < 0 || t > 1) return false;
00093 
00094   intersection->set(segment.a).add(segment.ray().scale(t));
00095 
00096   return true;
00097 }
