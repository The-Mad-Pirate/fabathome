00240 void OpenGLWidget::advanceCameraToTarget() {
00241 
00242   // smooth to next target
00243   FAHVector3 target = camera_->getTarget();
00244   target.add(next_target_.copy().sub(target).scale(0.1f));
00245   camera_->setTarget(target.x, target.y, target.z);
00246 
00247   // smooth to next eye
00248   FAHVector3 eye = camera_->getEye();
00249   FAHVector3 eye_dest;
00250   eye_dest.set(target).add(FAHVector3::make(150*zoom*sin(z_angle),150*zoom*cos(z_angle),150*zoom));
00251   eye.add(eye_dest.sub(eye).scale(0.05f));
00252   if (eye_dest.distanceToSq(eye) < 4.0f) {
00253     eye.set(eye_dest);
00254     advance_camera_timer_.stop();
00255   }
00256   camera_->setEye(eye.x, eye.y, eye.z);
