00062 //----[  runPathingAlgorithm  ]------------------------------------------------
00063 bool ConcentricLoopsPather::runPathingAlgorithm(
00064         const SliceMaterialRegion& region,
00065         PathSliceRegion* returned_paths,
00066         PatherProgressCallback* callback) {
00067   initialize();
00068 
00069   QVector<FAHLoopInXYPlane> boundaries;
00070   boundaries.append(region.getOuterBoundary());
00071   float last = 10000000.0f;
00072   while (boundaries.isEmpty() == false) {
00073     QVector<FAHLoopInXYPlane> contracted;
00074     FAHLoopInXYPlane front = boundaries.at(0);
00075     Path p(front);
00076     returned_paths->addPath(p);
00077     boundaries.remove(0);
00078     front.expand(-path_width_, &contracted);
00079     float e = contracted.at(0).areaEstimate();
00080     if (e > last) break;
00081     last = e;
00082     boundaries += contracted;
00083   }
00084 
00085   return true;
00086 }
