00131     for (int i = 0; i < mesh.size(); ++i) {
00132         const QList<AMFRegion*> *region = mesh[i]->GetRegions();
00133         QList<AMFTriangle*> tri;

00135             tri.append(region->at(j)->getTriangles());

00142             outTri.append(new AMFTriangle(tri.at(j)->getV1() + shiftVal,

00145     }

00148     for (int i = 0; i < 8; ++i) {
00149       outTri.append(new AMFTriangle(start + kTriangles[i*3+0],
00150                                     start + kTriangles[i*3+2],
00151                                     start + kTriangles[i*3+1]));
00152     }
00153 
00154     /*Generating output mesh*/
