00133 void MaterialsPanel::updateUI() {
00134   is_updating_ = true;
00135   QStringList compatible_script_names;
00136 
00137   { // Find all printers
00138     QSet<QString> printers;
00139     foreach (ToolScript* tool_script, tool_scripts_) {
00140       printers.insert(tool_script->printerName());
00141     }
00142     QString old = m_ui->printerCombo->currentText();
00143     int old_index = -1;
00144     m_ui->printerCombo->clear();

00146     for (int i = 0; i < printers_list.size(); ++i) {
00147       const QString& printer = printers_list.at(i);
00148       if ((old_index == -1) &&
00149           (0 == QString::compare(printer,old,Qt::CaseSensitive))) {
00150         old_index = i;
00151       }
00152       m_ui->printerCombo->addItem(printer);
00153     }
00154 
00155     // Select the first ool by default
00156     if (old_index == -1 && !printers_list.isEmpty()) old_index = 0;
00157 
00158     // Re-select a tool from the list
00159     m_ui->printerCombo->setCurrentIndex(old_index);
00160   }
00161 
00162   {
00163     QString current_printer_name = m_ui->printerCombo->currentText();
00164     QMutexLocker lock(&materials_mutex_);
00165 
00166     // All materials that can be assigned to objects are here
00167     QSet<QString> compatible_materials;
00168     QList<ToolScript*> compatible_scripts;
00169 
00170     // Search all of the scripts to find available materials
00171     foreach (ToolScript* tool_script, tool_scripts_) {
00172       QSet<QString> script_materials;
00173 
00174       // Ignore scripts that work with a different printer
00175       if (0 != QString::compare(tool_script->printerName(), current_printer_name)) {
00176         continue;
00177       }
00178 
00179       // Add all of the different processable materials
00180       foreach (ToolScriptTool* tool, tool_script->tools()) {
00181         script_materials.insert(tool->materialName());
00182       }
00183 
00184       bool is_compatible = true;
00185       foreach (const QString& material, assigned_materials_) {
00186         if (script_materials.contains(material) == false) {
00187           // this script can't be used since it isn't able to
00188           // process all of the materials in the scene
00189           is_compatible = false;
00190           break;
00191         }
00192       }
00193 
00194       if (is_compatible) {
00195         // If we get here, the script has all of the materials in the
00196         // scene and possibly extra--so add them all to the list of
00197         // available materials.
00198         compatible_materials.unite(script_materials);
00199         compatible_scripts.append(tool_script);
00200       }
00201     }
00202 
00203     // Update the set of compatible tool-scripts
00204     foreach (ToolScript* ts, compatible_scripts) {
00205       compatible_script_names.append(ts->name());
00206     }
00207 
00208     // Remove assigned materials from the compatible materials so the
00209     // user isn't confused by having the same material in multiple
00210     // lists.
00211     compatible_materials.subtract(assigned_materials_);
00212 
00213     // Reset the state of the compatible materials UI element and
00214     // re-fill it with the new set of material names.
00215     m_ui->compatibleMaterials->clear();
00216     foreach (const QString& material, compatible_materials) {
00217       QListWidgetItem* material_item = new QListWidgetItem();
00218       material_item->setText(material);
00219       m_ui->compatibleMaterials->addItem(material_item);
00220     }
00221 
00222     // Rebuild the state of the active materials UI element
00223     m_ui->assignedMaterials->clear();
00224     foreach (const QString& material, assigned_materials_) {
00225       QListWidgetItem* material_item = new QListWidgetItem();
00226       material_item->setText(material);
00227       m_ui->assignedMaterials->addItem(material_item);
00228 
00229       // Has this material been assigned a color?
00230       QColor assigned_color;
00231       if (!assigned_material_colors_.contains(material)) {
00232         // No color yet, so find a unique one and assign it
00233         static const Qt::GlobalColor kMaterialColorPalette[] = {
00234           Qt::green,     Qt::red,         Qt::yellow,
00235           Qt::cyan,      Qt::blue,        Qt::magenta,
00236           Qt::darkGreen, Qt::darkRed,     Qt::darkBlue,
00237           Qt::darkCyan,  Qt::darkMagenta, Qt::darkYellow
00238         };
00239         static const int kMaterialColorPaletteSize
00240             = sizeof(kMaterialColorPalette) / sizeof(Qt::GlobalColor);
00241         QList<QColor> used_colors = assigned_material_colors_.values();
00242         int i = 0;
00243         for (i = 0; i < kMaterialColorPaletteSize; ++i) {
00244           QColor color(kMaterialColorPalette[i]);
00245           if (!used_colors.contains(color)) {
00246             // assign this color to the material
00247             assigned_material_colors_.insert(material, color);
00248             assigned_color = color;
00249             break;
00250           }
00251         }
00252         if (i == kMaterialColorPaletteSize) {
00253           // We were unable to find a unique color, so just randomly pick a color
00254           // This could involve a guarantee of uniqueness by adding a while() loop
00255           // but it's unlikely that there will be more than a handful of materials
00256           // and any new color won't be highly distinguishable anyway, so adding the
00257           // code isn't really worth it.
00258           QColor color(qrand()%255, qrand()%255, qrand()%255);
00259           assigned_material_colors_.insert(material, color);
00260           assigned_color = color;
00261         }
00262       } else {
00263         assigned_color = assigned_material_colors_.find(material).value();
00264       }
00265 
00266       // Set the background to match the material's color so that it's easy
00267       // to match objects with materials
00268       material_item->setBackgroundColor(assigned_color);
00269     }
00270 
00271     { // Remove material color pairings for materials that are no longer assigned
00272       QList<QString> unused_materials;
00273       foreach (const QString& material, assigned_material_colors_.keys()) {
00274         if (!assigned_materials_.contains(material)) {
00275           unused_materials.append(material);
00276         }
00277       }
00278       foreach (const QString& material, unused_materials) {
00279         assigned_material_colors_.remove(material);
00280       }
00281     }
00282   }
00283 
00284   // Signal that the materials have changed.  This is done outside of
00285   // the tool-scripts mutex because it can cause deadlock otherwise.
00286   emit changedMaterials(compatible_script_names);
00287 
00288   // Finished!
00289   is_updating_ = false;
00290 }
