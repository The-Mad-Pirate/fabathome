00072 float FlexibleSettings::get(const QString& name, float default_value) {
00073   QMap<QString,SettingValue>::const_iterator i = values_.find(name);
00074   if (i == values_.end()) {
00075     return default_value;
00076   } else {
00077     const SettingValue& setting = i.value();
00078     switch (setting.type) {
00079       case SettingValue::FLOAT:   return setting.f;
00080       case SettingValue::INT:     return float(setting.i);
00081       case SettingValue::BOOL:    return setting.b ? 1.0f : 0.0f;
00082       default:
00083         { confirm(!"unknown setting type"); }
00084       case SettingValue::STRING: {
00085         float v;
00086         sscanf(setting.string.toAscii().data(), "%f", &v);
00087         return v;
00088       }
00089     }
00090   }
00091 }
