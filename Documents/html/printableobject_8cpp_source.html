00050 float snapParameterValue(float f, float multiple, float range) {
00051   float m = fmod(f, multiple);
00052   if (m > multiple/2) m -= multiple;
00053   float closest_ratchet = f - m;
00054   return (fabs(closest_ratchet - f) < range) ? closest_ratchet : f;
00055 }

00900 void PrintableObject::handleManipulateTranslation(PrintableObjectHandleAxis axis,
00901                                                   const FAHLine& start,
00902                                                   const FAHLine& end) {
00903 
00904   // Manipulation works as one would expect: when an axis is active, dragging along the
00905   // plane containing that axis causes the object to move only along that axis.
00906   // However, there is one exception.  Only having the horizontal-plane (x, y) handles
00907   // move in exactly their axes makes moving things around on the tray difficult.  So,
00908   // if the location of the mouse gets far from the straight axis, we "break" into
00909   // biaxial translation mode that allows the user to drag the object around on the xy
00910   // plane even though they are holding only either the x or y axes.
00911 
00912   const FAHFloat kBreakoutThreshold = 1;
00913 
00914   switch (axis) {
00915   case kXOnly:
00916   case kYOnly: {
00917     FAHPlane xy(bounding_sphere_.center, FAHVector3::make(0,0,1));
00918     FAHVector3 start_point(xy.intersectLine(start));
00919     FAHVector3 end_point(xy.intersectLine(end));
00920     manipulation_.translation.set(end_point).sub(start_point);
00921     manipulation_.translation.z = 0; // don't modify the Z-coordinate
00922 
00923     // Don't allow the other coordinate to be modified, unless the mouse has broken out
00924     if (axis == kXOnly) {
00925       FAHLine x(start_point, start_point.copy().add(FAHVector3::make(1,0,0)));
00926       if (x.lineDistanceTo2DXY(end_point) < kBreakoutThreshold) {
00927         manipulation_.translation.y = 0;
00928       }
00929     } else { // axis == kYOnly
00930       FAHLine y(start_point, start_point.copy().add(FAHVector3::make(0,1,0)));
00931       if (y.lineDistanceTo2DXY(end_point) < kBreakoutThreshold) {
00932         manipulation_.translation.x = 0;
00933       }
00934     }
00935   } break;
00936 
00937   case kZOnly: {
00938     // use a plane xz or yz depending on which is more perpendicular to the pick rays
00939     FAHPlane plane(bounding_sphere_.center,
00940                    fabs(start.ray().x)>0.5
00941                      ? FAHVector3::make(1,0,0)
00942                      : FAHVector3::make(0,1,0));
00943     FAHVector3 start_point(plane.intersectLine(start));
00944     FAHVector3 end_point(plane.intersectLine(end));
00945     manipulation_.translation.z = end_point.z - start_point.z;
00946   } break;
00947   }
00948 }

01009 PrintableObject::BoundingCubeFace PrintableObject::findBestOrientation() {
01010   BoundingCubeFace lowest_face;
01011   float lowest_height;
01012   for (int i = 0; i < 6; ++i) {
01013     BoundingCubeFace face_to_check = (BoundingCubeFace)i;
01014     FAHMatrix4x4 transform;
01015     transform.rotationEuler(getRotationToPutFaceDown(face_to_check));
01016 
01017     // Obtain a list of every single triangle in this object
01018     QList<FAHTriangle> triangles;
01019     foreach (const AMFObject* object, amf_->GetObjects()) {
01020       foreach (AMFRegion* region, *(object->GetMesh()->GetRegions())) {
01021         QList<FAHTriangle> region_triangles;
01022         region->getTriangles(&region_triangles);
01023         triangles += region_triangles;
01024       }
01025     }
01026 
01027     // Transform the set of triangles
01028     for (int t = 0; t < triangles.size(); ++t) {
01029       FAHTriangle& triangle = triangles[t];
01030       transform.transformInPlace(&triangle.v[0]);
01031       transform.transformInPlace(&triangle.v[1]);
01032       transform.transformInPlace(&triangle.v[2]);
01033     }
01034 
01035     // Find the lowest point of the mesh
01036     float min_z = triangles[0].v[0].z;
01037     for (int t = 0; t < triangles.size(); ++t) {
01038       FAHTriangle& triangle = triangles[t];
01039       min_z = min_z > triangle.v[0].z ? triangle.v[0].z : min_z;
01040       min_z = min_z > triangle.v[1].z ? triangle.v[1].z : min_z;
01041       min_z = min_z > triangle.v[2].z ? triangle.v[2].z : min_z;
01042     }
01043 
01044     // Use a Monte Carlo method to discover the center of gravity
01045     FAHVector3 center_of_gravity
01046         = calculateCenterOfGravity(triangles);
01047 
01048     // Determine if this center of gravity is better than others
01049     // (We have to adjust the center of gravity by the minimum
01050     //  z coordinate in order to compensate for rotation)
01051     float height = center_of_gravity.z - min_z;
01052     if (i == 0 || lowest_height > height) {
01053       lowest_height = height;
01054       lowest_face = face_to_check;
01055     }
01056   }
01057   return lowest_face;
