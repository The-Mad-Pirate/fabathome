00051 void OpenGLCamera::orbit(float mouseDeltaX, float mouseDeltaY) {
00052 
00053   GLVector3 negGaze;
00054   negGaze.set(eye);
00055   negGaze.sub(target);
00056   float dist = negGaze.length();
00057   negGaze.normalize();
00058 
00059   float azimuth = (float)atan2(negGaze.x, negGaze.y);
00060   float elevation = (float)atan2(negGaze.z, sqrt(negGaze.x * negGaze.x + negGaze.y * negGaze.y));
00061   azimuth = fmodf((azimuth + mouseDeltaX), (float) (2 * PI));
00062   elevation = (float) qMax(-PI * 0.495f, qMin(PI * 0.495f, (float)(elevation - mouseDeltaY)));
00063 
00064   negGaze.set((float) (sin(azimuth) * cos(elevation)),
00065               (float) (cos(azimuth) * cos(elevation)),
00066               (float) sin(elevation));
00067   negGaze.normalize();
00068 
00069   eye.set(target);
00070 
00071   GLVector3 temp;
00072   temp.set(negGaze.x, negGaze.y, negGaze.z);
00073   temp.scale(dist);
00074 
00075   eye.add(temp);
00076 
00077   updateFrame();
00078 }

00118 OpenGLCamera::GLVector3 OpenGLCamera::raycastFromViewport(float vx, float vy) const {
00119 
00120   // Build the vector of the pick ray in screen space
00121   GLVector3 v(vx / projection._11, vy / projection._22, -1.0f);
00122 
00123   // Convert to world-space
00124   // TODO: store a copy of the inverse view matrix when the camera is
00125   //       updated instead of recalculating it, because this calc is expensive
00126   GLMatrix4x4 m(view);
00127   m.invert();
00128 
00129   // The below calculation is an expansion of this commented code to take advantage
00130   // of known-value terms in the transformation.
00131   //m._41 = 0;  m._42 = 0;  m._43 = 0; // don't involve the camera's location
00132   //m.transformInPlace(&v);
00133   v.set(v.x*m._11 + v.y*m._21 + v.z*m._31,
00134         v.x*m._12 + v.y*m._22 + v.z*m._32,
00135         v.x*m._13 + v.y*m._23 + v.z*m._33);

00137   // Normalize the pick ray and return it
