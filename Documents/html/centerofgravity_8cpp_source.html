00103  /*
00104   * Calculate the center of gravity of the mesh of triangles
00105   * @param triangles (the mesh of triangles that represent the shape whose center of gravity calculateCoG finds)
00106   * @param tolerance (the tolerance of change between two consecutive center of gravity calculations)
00107   * @return cog (the estimated center of gravity for the triangle mesh)
00108   */
00109 FAHVector3 calculateCoG(const QList<FAHTriangle>& triangles,
00110                         double tolerance) {
00111 
00112     //accumulate as more center of gravity calculations are made
00113     FAHVector3 accumulator;
00114     //remember the previous center of gravity to check the tolerance
00115     FAHVector3 last;
00116     //used to check if last center of gravity is within tolerance of current center of gravity
00117     FAHVector3 temp;
00118     //the estimated center of gravity of the mesh
00119     FAHVector3 cog;
00120 
00121     //initialize variables
00122     last.zero();
00123     accumulator.zero();
00124     int counter=1;
00125     bool loop=true;
00126     MonteCarloReturnValues values;
00127 
00128     //continually calculate center of gravity until estimates of center of gravity remain within tolerance
00129     while (loop) {
00130 
00131         //store last CoG
00132         last.set(accumulator).scale(1.0/counter);
00133 
00134         //calculate current CoG
00135         monteCarlo(triangles, &values);
00136 
00137         accumulator.add(values.acc);
00138         counter+=values.counter;
00139 
00140         //see if last CoG is within tolerance of current CoG
00141         temp.set(accumulator).scale(1.0/counter);
00142         loop=(tolerance<temp.distanceTo(last));
00143     } //end while
00144 
00145     //average cog based off accumulator and the number of trials
00146     cog.set(accumulator).scale(1.0/counter);
00147 
00148     return cog;
00149 } //end calculateCoG

00153  * Randomly pick points inside the mesh and use these points to estimate the center of gravity

00162     //the maximum possible x,y,z bounds

00175     //store min/max x,y,z
00176     for(int i=0;i<triangles.length();i++) {
00177 
00178         maxBoundary.max(triangles.at(i).v[0]);
00179         maxBoundary.max(triangles.at(i).v[1]);
00180         maxBoundary.max(triangles.at(i).v[2]);
00181 
00182         minBoundary.min(triangles.at(i).v[0]);
00183         minBoundary.min(triangles.at(i).v[1]);
00184         minBoundary.min(triangles.at(i).v[2]);
00185 
00186     } //end for
00187

00189     centerPoint.set(maxBoundary.copy().add(minBoundary)).scale(1/2);
00190 
00191     //declare infinity
00192     FAHVector3 infinity;
00193     int counter=0;
00194 
00195     for(int i=0;i<300;i++) {
00196 
00197         //randomly find a point
00198         FAHVector3 randPoint;
00199         randPoint.set(((maxBoundary.copy().sub(minBoundary)).scale(qrand()/double(RAND_MAX))).add(minBoundary));
00200 
00201         //define infinity to be directly above the random point
00202         infinity.set(randPoint);

00204

00209 
00210         //check all nearby triangles to see how many the ray intersects
00211         for(int j=0;j<triangles.length();j++) {
00212 
00213             //the three points that make up the vertices of a triangle
00214             FAHVector3 p1;
00215             FAHVector3 p2;
00216             FAHVector3 p3;
00217 
00218             //calculate plausible triangles that the ray of your point to infinity could intersect
00219             p1.set(triangles.at(j).v[0]);

00221             p3.set(triangles.at(j).v[2]);
00222 
00223             //increment intersection counter if at least one vertex of the triangle has a z coordinate greater than the random point's z coordinate,
00224             //and if the ray and triangle intersect. For ease of reading, it is broken into 2 if statements.

00230                              intersectionCounter++;
