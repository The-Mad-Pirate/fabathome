00060 bool STLFile::read(QString path) {
00061 
00062   bool is_binary_file;
00063 
00064   QFile stl(path);
00065 
00066   if (!stl.open(QFile::ReadOnly)) {
00067     std::cout << "Could not open file." << std::endl;
00068   }
00069   QDataStream stream(&stl);
00070   QString line;
00071 
00072   if (stream.atEnd()) {
00073     // error
00074     std::cout << "There was an error reading the specified STL file.";
00075     return false;
00076   }
00077 
00078   // Detect ASCII vs. binary.
00079   // Read the last 20 characters of the file and
00080   // search for "endsolid" (indicates ASCII file).
00081   char buffer[20];
00082   stream.skipRawData(stl.size() - 19);
00083   stream.readRawData(buffer, 20);
00084   buffer[19] = '\0';
00085   line = buffer;
00086 
00087   stl.close();
00088   is_binary_file = !line.contains("endsolid", Qt::CaseInsensitive);
00089 
00090   return is_binary_file ? readBinary(path) : readASCII(path);
00091 }
