00092 int Sphere::intersectDistances(const Vector3& origin, const Vector3& ray,
00093                                float* enter_distance, float* exit_distance) const {
00094 
00095   confirm(ray.isNormalized()) else {
00096     if (floatsEqual(ray.length(), 0)) return 0;
00097     return intersectDistances(origin,
00098                               ray.copy().normalize(),
00099                               enter_distance,
00100                               exit_distance);
00101   }
00102 
00103   Vector3 distance;
00104   distance.set(origin).sub(center);
00105   Float b = distance.dot(ray);
00106   Float c = distance.magnitudeSq() - radius*radius;
00107   Float d = b*b - c;
00108   if (d > Float(0)) {
00109     d = sqrt(d);
00110     *enter_distance = -b - d;
00111     if (b > d) {
00112       // intersects twice; ray origin outside the sphere & pointing at sphere
00113       *exit_distance = -b + d;
00114       return 2;
00115     } else {
00116       // intersects once; ray origin leaving sphere or ray just barely skims surface
00117       *exit_distance = *enter_distance;
00118       return 1;
00119     }
00120   } else {
00121     return 0;
00122   }
00123 }
