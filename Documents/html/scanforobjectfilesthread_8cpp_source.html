00040 void ScanForObjectFilesThread::run() {
00041   QStringList directories;
00042   directories.append(QDir::currentPath());
00043   directories.append(QDir::homePath());
00044 
00045   while (!directories.isEmpty()) {
00046 
00047     // Pop the first directory from the worklist
00048     QDir directory = QDir(directories.first());
00049     directories.removeFirst();
00050 
00051     // Find the list of all object files in this directory
00052     QStringList files
00053         = directory.entryList(PrintableObject::loadableFileFilterExtensions(),
00054                               QDir::Files | QDir::NoSymLinks);
00055 
00056     // Pass this list to the load objects panel
00057     panel_->addDiscoveredObjectFiles(directory.absolutePath(), files);
00058 
00059     // Add subdirectories to worklist
00060     QStringList folders = directory.entryList(QStringList("*"), QDir::Dirs|QDir::NoDotAndDotDot);
00061     for (int i = 0; i < folders.size(); ++i) {
00062       directories.append(directory.absolutePath() + QDir::separator() + folders.at(i));
00063     }
00064   }
00065 }

00068 void ScanForObjectFilesThread::scanDirectory(const QString& path,
00069                                              const QStringList& files_to_ignore,
00070                                              LoadObjectsPanel* panel) {
00071 
00072   QDir directory = QDir(path);
00073   QString absolute_path = directory.absolutePath();
00074 
00075   // Find the list of all object files in this directory
00076   QStringList files
00077       = directory.entryList(PrintableObject::loadableFileFilterExtensions(),
00078                             QDir::Files | QDir::NoSymLinks);
00079 
00080   // Remove any entries that are already in the list
00081   for (int i = files.size() - 1; i >= 0; --i) {
00082     QString object_file_path = absolute_path + QDir::separator() + files.at(i);
00083     if (files_to_ignore.contains(object_file_path, Qt::CaseSensitive)) {
00084       files.removeAt(i);
00085     }
00086   }
00087 
00088   // Add the list
00089   panel->addDiscoveredObjectFiles(absolute_path, files);
00090 }
