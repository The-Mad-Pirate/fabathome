00035 //----[  LinkStaticTodoStructIfValid  ]----------------------------------------
00036 // Adds the TodoStruct ts to a linked list of structures local to this method,
00037 // as long as ts != NULL.
00038 // The head of the list is returned.
00039 TodoStruct* LinkStaticTodoStructIfValid(TodoStruct* ts) {
00040   static TodoStruct* first = 0;
00041   if (ts) {
00042     ts->next = first;
00043     first = ts;
00044   }
00045   return first;
00046 }

00111 TodoReport TodoReport::generateReport() {
00112   QString text;
00113   TodoStruct* next_todo = LinkStaticTodoStructIfValid(NULL);
00114   typedef std::vector<TodoStruct*> TodoStructs;
00115   TodoStructs all;
00116   while (next_todo) { // add every TodoStruct to the vector
00117     all.push_back(next_todo);
00118     next_todo = next_todo->next;
00119   }
00120 
00121   for (int report_type = 0; report_type < 2; ++report_type) {
00122     switch (report_type) {
00123     case 0:
00124       text.append("\n----[  todo() report by hits  ]----------------------------------\n");
00125       std::sort(all.begin(), all.end(), TodoStructSTLSortPredicateByHits);
00126       break;
00127     case 1:
00128       text.append("\n----[  todo() report by time  ]----------------------------------\n");
00129       std::sort(all.begin(), all.end(), TodoStructSTLSortPredicateByTime);
00130       break;
00131     }
00132     for (TodoStructs::iterator i = all.begin(); i != all.end(); ++i) {
00133       TodoStruct* ts = *i;
00134       QString line;
00135       QTextStream(&line)
00136         << ts->file
00137         << "(" << ts->line << "):  \""
00138         << ts->what
00139         << "\" - " << ts->who
00140         << " (hit " << ts->hits << " time" << (ts->hits==1 ? "":"s")
00141         << " - " << ts->local_ms << "ms)\n";
00142       text.append(line);
00143     }
00144   }
00145   text.append("-----------------------------------------------------------------\n");
00146 
00147   // allocate output buffer
00148   TodoReport tr;
00149   tr.text = text;
00150   return tr;
00151 }
