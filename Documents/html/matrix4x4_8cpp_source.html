00154 void Matrix4x4::transformInPlace(Vector3* v) const {
00155   // This is the 4th component of the resulting transformation.  The input vector's
00156   // 4th component is assumed to be 1.0 coming in, and the resulting vector is
00157   // normalized such that it would have a component of 1.0.
00158   Float w = _14 * v->x + _24 * v->y + _34 * v->z + _44;

00160     (_11 * v->x + _21 * v->y + _31 * v->z + _41) / w,
00161     (_12 * v->x + _22 * v->y + _32 * v->z + _42) / w,
00162     (_13 * v->x + _23 * v->y + _33 * v->z + _43) / w
