00103 void fabStudioDebugLogOutput(QtMsgType type, const char *msg) {
00104 
00105   switch (type) {
00106   case QtDebugMsg:
00107     debug_log.write("<pre><font color=\"black\">");
00108     debug_log.write(msg);
00109     debug_log.flush();
00110     break;
00111   case QtWarningMsg:
00112     debug_log.write("<pre><font color=\"orange\">");
00113     debug_log.write(msg);
00114     break;
00115   case QtCriticalMsg:
00116     debug_log.write("<pre><font color=\"red\">");
00117     debug_log.write(msg);
00118     break;
00119   case QtFatalMsg:
00120     debug_log.write("<pre><font color=\"red\">");
00121     debug_log.write(msg);
00122     break;
00123   }
00124   debug_log.write("</font></pre><br/>");
00125   if (type != QtDebugMsg) debug_log.flush();
00126   if (type == QtCriticalMsg ||
00127       type == QtFatalMsg) __asm("int3"); // let the debugger handle this problem!
00128   if (type == QtFatalMsg) abort();      // can't continue execution
00129 }
