00070 void PositionObjectsPanel::manipulatedObject(PrintableObject* object) {
00071   FAHVector3 t, r, s;
00072   if (object) {
00073     t.set(object->translation());
00074     r.set(object->rotation());
00075     s.set(object->scaling());
00076   } else {
00077     t.zero();
00078     r.zero();
00079     s.zero();
00080   }
00081   manipulating_ = true;
00082   m_ui->positionX->setValue(t.x);
00083   m_ui->positionY->setValue(t.y);
00084   m_ui->positionZ->setValue(t.z);
00085 #define toDeg(x) (x/Math::k2Pi)*360.0
00086   m_ui->rotationX->setValue(toDeg(r.x));
00087   m_ui->rotationY->setValue(toDeg(r.y));
00088   m_ui->rotationZ->setValue(toDeg(r.z));
00089   m_ui->scalingX->setValue(s.x*100);
00090   m_ui->scalingY->setValue(s.y*100);
00091   m_ui->scalingZ->setValue(s.z*100);
00092   manipulating_ = false;
00093 }
