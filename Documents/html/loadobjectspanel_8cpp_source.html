00121 
00122   // Add the number of files to the map

00124   if (objects_per_folder_.size() > kMaxWatchedFolders) {
00125     // eliminate the smallest one
00126     QString smallest_key = objects_per_folder_.begin().key();
00127     int smallest = objects_per_folder_.begin().value();
00128     QMap<QString, int>::iterator i = objects_per_folder_.begin();
00129     for (; i != objects_per_folder_.end(); ++i) {
00130       if (i.value() < smallest) {
00131         smallest_key = i.key();
00132         smallest = i.value();
00133       }
00134     }
00135     if (0 != QString::compare(smallest_key, path, Qt::CaseSensitive)) {
00136       // We removed an old path
00137       object_folders_watcher_.removePath(smallest_key);
00138       objects_per_folder_.remove(smallest_key);
00139     } else {
00140       objects_per_folder_.remove(path);
00141     }
00142   }
00143   if (objects_per_folder_.contains(path)) {
00144     object_folders_watcher_.addPath(path);
00145   }
00146 
00147   QString filter = m_ui->fileNameTypeahead->text();
00148   for (int i = 0; i < files.size(); ++i) {
00149     const QString& file_name = files.at(i);
00150 
00151     // Add the object to our internal list
00152     QString object_file_path = path + QDir::separator() + file_name;
00153     object_files_.append(object_file_path);
00154 
00155     // Only add files with names that match the current filter to the display list

00172 void LoadObjectsPanel::folderChanged(const QString& path) {
00173   if (scan_for_object_files_thread_) {
00174     // scanning has not finished; don't update otherwise we start
00175     // adding files like crazy.
00176     todo("kwg8", "re-scan changed directory while scan for object files thread is still running");
00177     return;
00178   }
00179 
00180   ScanForObjectFilesThread::scanDirectory(path,
00181                                           object_files_,
00182                                           this);
00183   
00184   // remove any files that disappeared
00185   bool file_deleted = false;
00186   for (int i = object_files_.size() - 1; i >= 0; --i) {
00187     const QString& object_file = object_files_.at(i);
00188     if (object_file.startsWith(path, Qt::CaseSensitive) &&
00189         !QFile::exists(object_file)) {
00190       object_files_.removeAt(i);
00191       file_deleted = true;
00192     } 
00193   }
00194 
00195   if (file_deleted) {
00196     // we have to re-sync the list
00197     on_fileNameTypeahead_textChanged(m_ui->fileNameTypeahead->text());
00198   }

00226 void LoadObjectsPanel::on_rescan_clicked() {
00227   // Disable this button to prevent multiple clicks
00228   m_ui->rescan->setEnabled(false);
00229 
00230   confirm (!scan_for_object_files_thread_) else return;
00231 
00232   // Stop watching all of the old folders
00233   foreach (const QString& folder, objects_per_folder_.keys()) {
00234     object_folders_watcher_.removePath(folder);
00235   }
00236   objects_per_folder_.clear();
00237   m_ui->objectsList->clear();
00238 
00239   // Establish a thread that searches the disk for object files and
00240   // lists them in the loading panel.  Connecting the finished() signal
